#include<sys/types.h>
#include<sys/socket.h>
#include<arpa/inet.h>
#include<netinet/in.h>
#include<iostream>
#include<unistd.h>
#include<string.h>
using namespace std;
#define PORT 6666
int main()
{
	int s_id=socket(AF_INET,SOCK_STREAM,0);
	char text[100];
	int option;
	if(s_id==-1)
	{
		cout<<"Socket could not be created "<<endl;
		exit(0);
		}
	struct sockaddr_in serv_addr;
	serv_addr.sin_family=AF_INET;
	serv_addr.sin_port=htons(PORT);
	serv_addr.sin_addr.s_addr=INADDR_ANY;
	for(int i=0;i<8;i++)
	serv_addr.sin_zero[i]=0;
	
	if(connect(s_id,(struct sockaddr *)&serv_addr,sizeof(serv_addr))==-1)
	{
		cout<<"Could not connect to server "<<endl;
		exit(0);
	}
	string flag="";
	do
	{
		cout<<"Enter the message : ";
		std::getline(std::cin,flag);
		for(int i=0;i<flag.size();i++)
		text[i]=flag.at(i);
		write(s_id,text,strlen(text));
		if(read(s_id,text,100)!=-1)
		cout<<"Server : "<<text<<endl;	
		}while(flag!="stop");
	//if(n!=-1)
	//cout<<"Server : "<<text<<endl;
	
}
