#include<iostream>
#include<sys/socket.h>
#include<sys/types.h>
#include<netinet/in.h>
#include<unistd.h>
#define PORT 6666
using namespace std;
int main()
{
	int sock_fd=socket(AF_INET,SOCK_DGRAM,0);
	if(sock_fd<0)
	{
		cout<<"Socket Couldn't be created "<<endl;
		exit(0);
	}
	struct sockaddr_in serv_addr;
	serv_addr.sin_family=AF_INET;
	serv_addr.sin_port=htons(PORT);
	serv_addr.sin_addr.s_addr=INADDR_ANY;
	
	char text[30]="Hello Server ";
	int return_value=sendto(sock_fd,text,sizeof(text),0,(struct sockaddr *)&serv_addr,sizeof(serv_addr));
	if(return_value<0)
	cout<<"Send Unsuccesfull "<<endl;
	else
	cout<<"Send Successfull "<<endl;
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	



}
